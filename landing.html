stdio IDE Integration

# Add to .cursor/mcp.json
{
  "mcpServers": {
    "jebat": {
      "command": "python",
      "args": ["-m", "jebat.features.mcp.mcp_server"]
    }
  }
}

HTTP Remote Access

# Start HTTP transport
jebat mcp serve --transport http --port 8099

# Streamable HTTP (MCP 2025-03-26)
jebat mcp serve --transport streamable-http --port 8100

# List available tools
jebat mcp serve --help

Cli Quick Start

# Start as stdio server (for IDE)
jebat mcp serve --transport stdio

# Generate IDE config templates
jebat mcp serve --print-ide-config

npx Zero-Install MCP

# Via npx (no install required)
npx @nusabyte/jebat mcp serve

# Connect external MCP server
jebat mcp connect --http http://host:8100/mcp

Quick Start

Start in 30 seconds

npx Zero Install

# Interactive REPL
npx @nusabyte/jebat repl

# One-shot chat
npx @nusabyte/jebat chat "Explain memory consolidation"

# With bun
bunx @nusabyte/jebat repl

pip Global Install

# Via pip
pip install jebat
jebat repl

# Via npm global
npm install -g @nusabyte/jebat
jebat repl

MCP IDE Integration

{
  "mcpServers": {

    }
  }
}

git Source Build

# Clone & install
git clone https://github.com/nusabyte-my/jebat-core.git
cd jebat-core
pip install -e .
