example.py 
from jebat.features.cortex import Cortex
from jebat_dev.brain import DevBrain

# Initialize Cortex reasoning engine
cortex = Cortex(max_thoughts=20)

# Deep reasoning on complex problems
result = await cortex.think(
    problem="Optimize database queries",
    mode=ThinkingMode.DELIBERATE,
)

# Use DevAssistant for code generation
brain = DevBrain()
await brain.execute_task(
    task_type="scaffold",
    project_name="myapp",
    project_type="python_package",
)
