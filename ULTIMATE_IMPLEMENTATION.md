```bash
curl -X POST http://localhost:8000/api/v1/chat/completions \
  -H "Content-Type: application/json" \
  -d '{
    "message": "What is artificial intelligence?",
    "mode": "deliberate",

  }'
```

### 2. Use Python SDK

```python
from jebat_sdk import JEBATClient

async with JEBATClient() as client:
    # Chat
    response = await client.chat("What is AI?")
    print(f"Response: {response.response}")
    print(f"Confidence: {response.confidence:.1%}")
    
    # Store memory
    await client.store_memory(
        "I prefer Python for development",
        user_id="user123"
    )
    
    # Search memories
    memories = await client.search_memories(
        "Python",
        user_id="user123"
    )
    
    # Get metrics
    metrics = await client.metrics()
    print(f"Ultra-Loop cycles: {metrics.ultra_loop.get('total_cycles')}")
```

### 3. Use JavaScript SDK

```typescript
import { JEBATClient } from '@jebat/sdk';

const client = new JEBATClient();

// Chat
const response = await client.chat('What is AI?');
console.log(response.response);

// Store memory
await client.storeMemory('I like TypeScript', { userId: 'user123' });

// Search
const memories = await client.searchMemories('TypeScript', { 
  userId: 'user123' 
});
```

### 4. Multi-Tenancy

```python
from jebat.multitenancy import TenantManager, PlanType

manager = TenantManager()

# Create tenant
tenant = await manager.create_tenant(
    name="acme-corp",
    display_name="Acme Corporation",
    plan=PlanType.PRO,
)

# Set quotas
await manager.set_quota(tenant.id, "api_calls_per_day", 100000)

# Check quota
available = await manager.check_quota(tenant.id, "api_calls_per_day")

# Consume quota
await manager.consume_quota(tenant.id, "api_calls_per_day", 1)

# Get usage
usage = await manager.get_usage_summary(tenant.id, period="day")
```

### 5. Analytics

```python
from jebat.analytics import AnalyticsEngine

engine = AnalyticsEngine()

# Track events
await engine.track_event(
    "chat_completion",
    metadata={"duration": 1.5, "model": "gpt-4"},
    user_id="user123",
)

# Get insights
insights = await engine.get_insights(period="day")

# Get usage report
report = await engine.get_usage_report(period="week")

# Get user analytics
user_analytics = await engine.get_user_analytics("user123")
```

### 6. Plugin System

```python
from jebat.plugins import PluginManager

manager = PluginManager(plugins_dir=Path("plugins"))

# Discover plugins
plugins = manager.discover_plugins()

# Load plugin

    "my_plugin",
    config={"api_key": "secret"}
)

# Execute plugin
result = await manager.execute_plugin("my_plugin", {"data": "value"})

# List plugins
plugin_list = manager.list_plugins()
```
